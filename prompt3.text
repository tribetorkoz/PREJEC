## MULTI-TENANT ROUTING:
# Each company gets their own Twilio phone number
# Route incoming calls to correct company agent
# Method: match Twilio 'To' number to company in database

async def route_call_to_company(to_number: str):
    company = await db.get_company_by_phone(to_number)
    if not company:
        # Play: 'This number is not in service'
        return
    agent = await db.get_active_agent(company.id)
    return agent
