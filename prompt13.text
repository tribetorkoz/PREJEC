```python
# FILE: apps/api/src/features/voice/pipeline.py
# CURRENT: Sequential processing (slow)
# UPGRADE: Parallel processing (fast)

# REPLACE the current pipeline with this architecture:

import asyncio
from pipecat.pipeline.parallel_pipeline import ParallelPipeline

class UltraLowLatencyPipeline:
    """
    Process STT + LLM context loading in parallel.
    While Deepgram converts speech to text,
    simultaneously load customer memory and CRM data.
    This cuts 200-300ms from every response.
    """
    
    async def process(self, audio_frame):
        # Run these THREE things at exactly the same time:
        stt_task = asyncio.create_task(
            self.deepgram.transcribe(audio_frame)
        )
        memory_task = asyncio.create_task(
            self.load_customer_memory(self.caller_phone)
        )
        crm_task = asyncio.create_task(
            self.load_crm_context(self.caller_phone)
        )
        
        # Wait for ALL to complete simultaneously
        transcript, memory, crm_data = await asyncio.gather(
            stt_task, memory_task, crm_task
        )
        
        # LLM already has context loaded — responds instantly
        response = await self.llm.respond(
            transcript=transcript,
            context=f"{memory}\n{crm_data}"
        )
        
        # Start TTS immediately, stream audio as it generates
        async for audio_chunk in self.tts.stream(response):
            await self.transport.send_audio(audio_chunk)
```

### 1.2 Audio Streaming (Don't wait — stream)

```python
# FILE: apps/api/src/features/voice/streaming.py
# UPGRADE: Stream TTS audio as it generates
# Instead of waiting for full response, send audio chunks immediately

class AudioStreamingEngine:
    """
    Start playing audio after first 50ms of generation.
    ElevenLabs supports streaming — use it.
    This makes responses feel instant.
    """
    
    async def stream_response(self, text: str, transport):
        # Split text into sentences for faster first audio
        sentences = self.split_into_sentences(text)
        
        tasks = []
        for i, sentence in enumerate(sentences):
            # Generate and send each sentence as soon as ready
            task = asyncio.create_task(
                self.generate_and_send(sentence, transport, priority=i)
            )
            tasks.append(task)
        
        # First sentence plays in ~150ms
        await asyncio.gather(*tasks)
    
    def split_into_sentences(self, text: str) -> list:
        # Split on . ! ? for natural audio chunks
        import re
        return [s.strip() for s in re.split(r'[.!?]', text) if s.strip()]
```

```python
# FILE: apps/api/src/features/voice/cache.py
# Cache common answers — respond in <50ms for FAQ

import redis
import hashlib

class VoiceResponseCache:
    """
    Cache AI responses for common questions.
    "What are your hours?" → cached, responds in 50ms not 800ms
    80% of calls ask the same 20 questions.
    """
    
    async def get_cached_response(
        self, 
        question: str, 
        company_id: str
    ) -> str | None:
        cache_key = f"voice:{company_id}:{hashlib.md5(question.encode()).hexdigest()}"
        return await self.redis.get(cache_key)
    
    async def cache_response(
        self, 
        question: str, 
        response: str, 
        company_id: str,
        ttl: int = 3600  # 1 hour
    ):
        cache_key = f"voice:{company_id}:{hashlib.md5(question.encode()).hexdigest()}"
        await self.redis.setex(cache_key, ttl, response)
    
    # Auto-populate cache from most common questions
    async def warm_cache(self, company_id: str, faq: dict):
        for question, answer in faq.items():
            await self.cache_response(question, answer, company_id)
```

```python
from enum import Enum

class EmotionState(Enum):
    HAPPY = "happy"
    NEUTRAL = "neutral"
    CONFUSED = "confused"
    FRUSTRATED = "frustrated"

class RealTimeEmotionEngine:
    """
    Detect customer emotion from:
    1. Words used (sentiment analysis)
    2. Speaking pace (fast = frustrated/urgent)
    3. Volume patterns (loud = angry)
    4. Repeated questions (confused)
    
    Adapt agent personality in real-time.
    No competitor does all 4 signals together.
    """
    
    EMOTION_SIGNALS = {
        EmotionState.ANGRY: {
            "keywords": ["ridiculous", "unacceptable", "terrible", 
                        "worst", "lawsuit", "never coming back"],
            "response_style": "ultra_calm_empathetic",
            "pace": "slow",
            "escalate_after_turns": 2
        },
        EmotionState.URGENT: {
            "keywords": ["emergency", "urgent", "right now", "immediately",
                        "bleeding", "broken", "flooding", "pain"],
            "response_style": "fast_efficient_reassuring",
            "pace": "match_customer",
            "escalate_after_turns": 0  # Immediate escalation
        },
        EmotionState.CONFUSED: {
            "keywords": ["what do you mean", "i don't understand",
                        "can you repeat", "what?", "huh?"],
            "response_style": "simple_clear_patient",
            "pace": "slow",
            "use_examples": True
        },
        EmotionState.FRUSTRATED: {
            "keywords": ["already told you", "third time", "keep saying",
                        "not working", "still", "again"],
            "response_style": "empathetic_solution_focused",
            "pace": "moderate",
            "acknowledge_first": True
        }
    }
    
    def detect_emotion(self, transcript: str, 
                       speaking_pace: float,
                       volume_level: float) -> EmotionState:
        # Combine all signals for accurate detection
        text_emotion = self._analyze_text(transcript)
        pace_signal = self._analyze_pace(speaking_pace)
        volume_signal = self._analyze_volume(volume_level)
        
        # Weighted combination
        return self._combine_signals(
            text_emotion, pace_signal, volume_signal
        )
    
    def get_adapted_system_prompt(
        self, 
        base_prompt: str,
        emotion: EmotionState,
        turn_count: int
    ) -> str:
        style = self.EMOTION_SIGNALS.get(emotion, {})
        
        if not style:
            return base_prompt
        
        # Inject emotion-specific instructions
        adaptation = f"""
        
        CURRENT CUSTOMER EMOTION: {emotion.value}
        ADAPT YOUR RESPONSE:
        - Style: {style.get('response_style')}
        - Pace: {style.get('pace')}
        - Turn count: {turn_count}
        - If ANGRY/URGENT: transfer to human after {style.get('escalate_after_turns')} turns
        """
        
        return base_prompt + adaptation
```

## 3.2 Circuit Breaker Pattern

```python
# FILE: apps/api/src/core/circuit_breaker.py

class CircuitBreaker:
    """
    If a provider fails 3 times in 60 seconds,
    stop using it for 5 minutes automatically.
    Prevents cascade failures.
    Used by Netflix and Amazon for 99.99% uptime.
    """
    
    def __init__(self, failure_threshold=3, timeout=300):
        self.failure_threshold = failure_threshold
        self.timeout = timeout
        self.failures = {}
        self.state = {}  # CLOSED, OPEN, HALF_OPEN
    
    async def call(self, provider_name: str, func, *args):
        if self.state.get(provider_name) == "OPEN":
            if self.should_try_reset(provider_name):
                self.state[provider_name] = "HALF_OPEN"
            else:
                raise Exception(f"{provider_name} circuit is OPEN")
        
        try:
            result = await func(*args)
            self.on_success(provider_name)
            return result
        except Exception as e:
            self.on_failure(provider_name)
            raise e
```

```python
class PCIComplianceHandler:
    """
    Handle credit card payments over voice calls.
    Pause recording during card number input.
    Mask card numbers in transcripts automatically.
    Essential for any business taking payments.
    """
    
    async def handle_payment_flow(self, call_sid: str):
        # Pause recording for PCI compliance
        await self.twilio.pause_recording(call_sid)
        
        # Agent collects card number using DTMF tones (keypad)
        # No voice recording of card numbers
        card_data = await self.collect_dtmf_input(
            call_sid,

            expected_length=16,
            masked=True
        )
        
        # Process payment through Stripe
        result = await self.stripe.charge(card_data)
        
        # Resume recording after payment
        await self.twilio.resume_recording(call_sid)
        
        return result
    
    async def mask_card_in_transcript(self, transcript: str) -> str:
        # Auto-mask any card-like patterns
        import re
        masked = re.sub(
            r'\b\d{4}[\s-]?\d{4}[\s-]?\d{4}[\s-]?\d{4}\b',
            '[CARD MASKED]',
            transcript
        )
        return masked
```

```python
        history = await self.db.get_cross_channel_history(
            customer_identifier, 
            company_id
        )
        
        # Build context from all previous interactions
        context = self.build_omnichannel_context(history)
        
        # Route to appropriate handler
        if channel == "phone":
            return await self.handle_voice(
                message, context, company_id, customer_identifier
            )
        elif channel in ["whatsapp", "sms"]:
            return await self.handle_text(
                message, context, company_id, customer_identifier
            )
        elif channel == "webchat":
            return await self.handle_webchat(
                message, context, company_id, customer_identifier
            )
    
    async def get_cross_channel_history(
        self, 
        customer_id: str,
        company_id: str
    ) -> list:
        return await self.db.query("""
            SELECT channel, message, response, timestamp
            FROM interactions
            WHERE customer_id = ? AND company_id = ?
            ORDER BY timestamp DESC
            LIMIT 20
        """, [customer_id, company_id])
```

```python
# FILE: apps/api/src/features/sales/roi_calculator.py

class ROICalculator:
    """
    Show every prospect exactly how much money
    they'll save and make with VoiceCore.
    Data-driven sales — closes deals faster.
    """
    
    def calculate(
        self,
        monthly_calls: int,
        current_receptionist_salary: float,
        missed_call_rate: float,
        avg_customer_value: float,
        plan_price: float
    ) -> dict:
        
        # Cost savings
        receptionist_cost = current_receptionist_salary
        voicecore_cost = plan_price
        monthly_savings = receptionist_cost - voicecore_cost
        
        # Revenue recovered from missed calls
        missed_calls_per_month = monthly_calls * (missed_call_rate / 100)
        calls_now_answered = missed_calls_per_month * 0.95  # 95% answer rate
        
        # Assume 30% of answered missed calls convert
        new_customers_per_month = calls_now_answered * 0.30
        monthly_revenue_recovered = new_customers_per_month * avg_customer_value
        
        # Total ROI
        total_monthly_benefit = monthly_savings + monthly_revenue_recovered
        monthly_roi_percent = ((total_monthly_benefit - plan_price) / plan_price) * 100
        payback_days = plan_price / (total_monthly_benefit / 30)
        
        return {
            "monthly_cost_savings": round(monthly_savings, 2),
            "monthly_revenue_recovered": round(monthly_revenue_recovered, 2),
            "total_monthly_benefit": round(total_monthly_benefit, 2),
            "annual_benefit": round(total_monthly_benefit * 12, 2),
            "roi_percent": round(monthly_roi_percent, 1),
            "payback_period_days": round(payback_days, 0),
            "summary": f"${round(total_monthly_benefit):,} benefit per month for ${plan_price} investment"
        }
```
